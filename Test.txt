INF = 9999999
V = 5
# Adjacency matrix for the graph
G = [[0, 9, 75, 0, 0],
     [9, 0, 95, 19, 42],
     [75, 95, 0, 51, 66],
     [0, 19, 51, 0, 31],
     [0, 42, 66, 31, 0]]

selected = [False] * V  # Track selected vertices
selected[0] = True  # Start with the first vertex
no_edge = 0

print("Edge : Weight")
while no_edge < V - 1:
    minimum = INF
    x = y = 0
    for i in range(V):
        if selected[i]:
            for j in range(V):
                if not selected[j] and G[i][j]:  # Check adjacent vertices
                    if G[i][j] < minimum:
                        minimum = G[i][j]
                        x, y = i, j
    print(f"{x}-{y}: {G[x][y]}")
    selected[y] = True
    no_edge += 1
